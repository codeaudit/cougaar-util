Properties file for controlling the COUGAAR Node Server.

Property files should usually end with a ".props" extension.  The
syntax is trivial and is defined by the JDK Properties.load() method.

Server control properties (with defaults given)
# The port of the server's RMI Registry
# (do not change)
org.cougaar.tools.server.port=8484

# The name of the server in the RMI Registry at org.cougaar.tools.server.port 
# (do not change)
org.cougaar.tools.server.name=NodeServer

# The class of RemoteNodeServer to instantiate 
# (do not change)
org.cougaar.tools.server.class=org.cougaar.tools.server.RemoteNodeServerImpl

# when true, the server will report on various activities
org.cougaar.tools.server.verbose=false

# The class of Node to start
# (do not change)
org.cougaar.tools.server.node.class=org.cougaar.core.society.Node

# The java vm command to run nodes with
org.cougaar.tools.server.java=java

# some standard memory settings
org.cougaar.tools.server.vm.memory=-Xms100m -Xmx300m

# use green threads?
# should be true for jdk1.2 under linux
org.cougaar.tools.server.vm.green=false

# The classpath to use to run nodes - defaults to the classpath of the 
# server environment. Note that only core.jar needs to be in here.
java.class.path=.../core.jar

# where cougaar is installed - no default
# This should be set to the value of $COUGAAR_INSTALL_PATH
org.cougaar.install.path=

# may specify additional -D arguments
org.cougaar.vm.arguments=

In addition, the console will send the following properties to the
server in order to start the Node 

# which config directory to use, e.g. integ-config
# no default, required.
org.cougaar.config=

# Which node of a config to start - e.g. SingleNode
# no default, required
org.cougaar.node.name=

# additional arguments to Node
# no default, optional
org.cougaar.node.arguments=

# Communications server and ports.
# Defines where the nameserver will run and which port the
# PSP webserver will run on.
# format is "hostname:nameserverport:PSPport"
# example: "alp42:9999:6666"
# The port numbers should be > 1000 and must not conflict with any
# other nodes running on the same set of computers.
# The standard alpserver clients provide this value automatically.
# if no value is passed, the nodes started will fall back to 
# using the alpreg.ini config file.
org.cougaar.name.server=
